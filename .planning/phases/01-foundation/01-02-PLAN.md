import * as SQLite from 'expo-sqlite';

Create src/storage/index.ts — unified re-export:

  <name>Task 2: Build instant-launch home screen shell</name>

  <action>
Replace the placeholder in app/index.tsx with the actual shell UI. Key constraints:
- Zero async operations on mount — no database queries, no storage reads, no network calls
- Dark background (#0a0a0a) — reduces visual overwhelm, ADHD-friendly
- Single clear prompt + single large CTA — nothing to interpret, nothing to decide
- StatusBar hidden — maximizes screen space, reduces distraction
- Button navigates to /result as a temporary placeholder (Phase 2 replaces this with the craving list)

```tsx
import { View, Text, TouchableOpacity, StatusBar, StyleSheet } from 'react-native';
import { router } from 'expo-router';

export default function HomeScreen() {
  return (
    <View style={styles.container}>
      <StatusBar hidden />
      <Text style={styles.headline}>What are you{'\n'}craving?</Text>
      <TouchableOpacity
        style={styles.cta}
        onPress={() => router.push('/result')}
        activeOpacity={0.8}
      >
        <Text style={styles.ctaText}>I'm craving...</Text>
      </TouchableOpacity>
    </View>
  );
}

const styles = StyleSheet.create({
  container: {
    flex: 1,
    backgroundColor: '#0a0a0a',
    justifyContent: 'center',
    alignItems: 'center',
    paddingHorizontal: 24,
  },
  headline: {
    fontSize: 36,
    fontWeight: '700',
    color: '#ffffff',
    textAlign: 'center',
    marginBottom: 48,
    lineHeight: 46,
  },
  cta: {
    backgroundColor: '#6366f1',

    paddingHorizontal: 40,
    borderRadius: 16,
    width: '100%',
    alignItems: 'center',
  },
  ctaText: {
    color: '#ffffff',
    fontSize: 20,
    fontWeight: '600',
  },
});
```
